Excercise

Get the first odd/even numbers


SOLUTION

Use:

 ses4.OddEvenNumbers(n);
ses4.OddEvenNumbers(6);

Output:

 The fist 6 odd numbers are:1,3,5,7,9,11
The first 6 even numbers are:2,4,6,8,10,12

Code:

 
var getOddNumber = function (range)
{
    var oddNumbers = [];
    for(var i = 1; i <= range * 2; i++)
    {
        if((i % 2) == 0)
        {
            continue;
        }
        oddNumbers.push(i);
    }
    return oddNumbers;
};

var getEvenNumber = function (range)
{
    var evenNumbers = [];
    for(var i = 1; i <= range * 2; i++)
    {
        if((i % 2) != 0)
        {
            continue;
        }
        evenNumbers.push(i);
    }
    return evenNumbers;
};

ses4.prototype.OddEvenNumbers = function(range)
{
    console.log('The fist '+ range + ' odd numbers are:' + getOddNumber(range));
    console.log('The first '+ range + ' even numbers are:' + getEvenNumber(range));
};
